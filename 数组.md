#### char*, char[] 和 string

+ 参数传递
有时候在传递参数的时候会发生疑惑，事实证明，char[] 和 char*是一样的
而且在作为参数传递的时候，指针传递，主函数里面定义的是指针，调用函数的时候直接写入即可
主函数里面如果是变量，则要取地址符&
传递进入reverse函数中的变量str和在主函数中使用并没有任何区别

+ 元素的调用
* 和 [] 的作用是一样的， 都有取元素的功能

一个一维int数组，数组名是一个元素地址
解引用是第一个元素

一个char 数组，数组名是整个字符串
解引用是第一个元素


```
#include<iostream>
#include<string>
#include<vector>
using namespace std;

//参数传递的时候 char str[] 和 char *str 一样的都是指针
void reverse(const char str1[]){
	cout << "reverse: " << endl;
	cout << endl;
	cout << "str1: " << str1 << endl;   //hello
	cout << "*str1: " << *str1 << endl;  // h
	cout << "str1[0]: " << str1[0] << endl; // h
	cout << "&str1: " << &str1 << endl; //010FFD58
	cout << "sizeof(str1): " << sizeof(str1) << endl;
	cout << "strlen(str1): " << strlen(str1) << endl;
	cout << "str1 + 1: " << str1 + 1 << endl;
	cout << "*(str1 + 1): " << *(str1 + 1) << endl;
}

void reverse2(const char *str1){

	//和在外部使用时一样的
	cout << "reverse2: " << endl;
	cout << "********" << endl;
	cout << "str1: " << str1 << endl;   //hello
	cout << "*str1: " << *str1 << endl;  // h
	cout << "str1[0]: " << str1[0] << endl; // h
	cout << "&str1: " << &str1 << endl; //010FFD58
	cout << "sizeof(str1): " << sizeof(str1) << endl;
	cout << "strlen(str1): " << strlen(str1) << endl;
	cout << "str1 + 1: " << str1 + 1 << endl;
	cout << "*(str1 + 1): " << *(str1 + 1) << endl;
}


int main(){
	char *str = "hello"; //本身就是指针
	cout << "typeid str: " << typeid(str).name() << endl; //char*
	cout <<"str: "<< str << endl;
	cout << "*str: " << *str << endl; //解引用成为第一个元素
	cout << "str[0]: " << str[0] << endl; //姐引用成为第一个元素
	cout << "&str: "<<&str << endl;
	cout << "sizeof(str): " << sizeof(str) << endl; //4 一个指针的大小
	cout << "strlen(str): " << strlen(str) << endl;
	cout << "str + 1: " << str + 1 << endl;
	cout << "*(str + 1): " << *(str + 1) << endl;

	char s = 's'; //单个字符

	cout << "----------------------------------------" << endl;
	char str1[] = "hello";
	cout << "typeid str1: "<<typeid(str1).name() << endl;  // char [6]

	cout << "str1: " << str1 << endl;   //hello  
	cout << "*str1: " << *str1 << endl;  // h
	cout << "str1[0]: " << str1[0] << endl; // h
	cout << "&str1: " << &str1 << endl; //010FFD58
	cout << "sizeof(str1): " << sizeof(str1) << endl; //6   整个数组的大小
	cout << "strlen(str1): " << strlen(str1) << endl; //5  strlen()  变量必须是 char* 类型的
	cout << "str1 + 1: " << str1 + 1 << endl;  // ello
	cout << "*(str1 + 1): " << *(str1 + 1) << endl; // e

	cout << "----------------------------------------" << endl;

	reverse(str);

	cout << "----------------------------------------" << endl;
	reverse2(str1);  //数组名会退化为指针


 //	reverse2(&str1)  error  str1本身就是地址了，不可以取地址符
	// reverse2(&s);
//  reverse(s);  error   因为需要传入地址


	//string 里面的每一个字符是char类型的；
	string s2 = "hello";
	cout << typeid(s2[0]).name() << endl; //char
	//s[0] = "h"; //错误
	s2[0] = ' ';   //需要用字符方式赋值
	cout << typeid(s2.data()).name() << endl;  //char const *

	char s1[] = "hello";
	//s1 ="hello"; //错误，因为字符串常量，不是可修改的左值

}
```
